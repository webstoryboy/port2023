안녕하세요! 웹스토리보이입니다. 자바스크립트 데이터 저장하기를 제대로 배웠다면 테스트를 한번 해봐야 겠죠! 80점 이상 나왔다면 잘 이해하고 있는 겁니다. 만약 그 점수가 나오지 않았다면 한번 더 공부해주세요! 🥳

1. 데이터 불러오기

01. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, "javascript"];

        console.log(arr[0], arr[1], arr[2]);
}

02. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

        for(let i=0; i<arr.length; i++){
                console.log(arr[i]);
        }
}

03. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

        arr.forEach(el => console.log(el));
}

04. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, 300, 400, 500];

        for(let element of arr){
                console.log(element);
        }    
}

05. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200];

        for(let element in arr){
            console.log(element + " : " + arr[element]);
        }
}

06. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, 300, 800, 900];

        arr.map(function(element, index){
                console.log(index);
        });
}

07. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

        const arrNum = arr.filter((el) => {
                return el >= 300;
        });

        console.log(arrNum);
}

08. 다음을 보고 결괏값을 작성하시오! 

{
        const arr1 = [100, 200, 300];
        const arr2 = [400, 500, 600];

        console.log(arr1);                  
        console.log(...arr1);              
        console.log(arr1 + arr2);         
        console.log(...arr1, ...arr2);     
        console.log(...arr1, 400);       
        console.log(...arr1.slice(0,2));
}

09. 다음을 보고 결괏값을 작성하시오! 

{
        function addNum(a, b, c) {
                return a + b + c;
        }
        const num = [100, 250, 350];
        const sum = addNum(...num);

        console.log(sum); 
}

10. 다음을 보고 결괏값을 작성하시오! 

{
        const [a, b, c] = [100, 200, 300];

        console.log(a);
        console.log(b);
        console.log(c);
}

11. 다음을 보고 결괏값을 작성하시오! 

{
        const arr = [100, 200, 300];
        const [x, y, z] = arr;

        console.log(x);
        console.log(y);
        console.log(z);
}

12. 다음을 보고 결괏값을 작성하시오! 

{
        const num = [100, 200, 300, 400, 500, 600];
        const [num1, num2, ...rest] = num;

        console.log(num1);
        console.log(num2);
        console.log(...rest);
        console.log(rest);
}

17. 다음을 보고 결괏값을 작성하시오!

18. 다음을 보고 결괏값을 작성하시오! 

{
        const obj = {
                a: 100,
                b: 200,
                c: "javascript"
        }
        for(let el in obj){
                console.log(el + " : " + obj[el]);
        }
}

19. 다음을 보고 결괏값을 작성하시오! 

{
        const obj1 = {
                a: 100,
                b: 200,
                c: "javascript",
        }
        const obj2 = {
                d: 300,
                e: 400,
                f: "jquery",
        }
        const obj3 = {...obj1, ...obj2}
        const obj4 = {...obj1, d: "jquery"}
        const obj5 = {...obj1, b: 300}

        console.log(obj3); 
        console.log(obj4); 
        console.log(obj5); 
}

20. 다음을 보고 결괏값을 작성하시오! 

{
        const obj1 = {
                a: 100,
                b: 200,
                c: "javascript",
        }
        const { a, b, c } = obj1;

        console.log(a);
        console.log(b);
        console.log(c);

        const { a: x, b: y, c: z } = obj1;

        console.log(x);
        console.log(y);
        console.log(z);

        const obj2 = {
                d: 100,
                e: 200,
                f: "javascript",
        }

        const { d, e, f, g = "jquery"} = obj2;

        console.log(g);

        const obj3 = {
                x1: 100,
                y1: {a1: 100, b1: 200},
                z1: "javascript",
        }
        const { x1, y1: {a1, b1}, z1} = obj3;

        console.log(x1);
        console.log(a1);
        console.log(b1);
        console.log(z1);
}
