1. 데이터 저장하기

01. 다음을 보고 결괏값을 작성하시오! 

{
        let x = 100;            
        let y = 200;            
        let z = "javascript";   

        console.log(x);     
        console.log(y);     
        console.log(z);     
}

02. 다음을 보고 결괏값을 작성하시오! 

{
        let x = 100;           
        let y = 200;            
        let z = "javascript";   

        x -= 300;        
        y += 400;       
        z = "jquery";   

        console.log(x);   
        console.log(y);   
        console.log(z);       
}

03. 다음을 보고 결괏값을 작성하시오! 

{

        arr[2] = "javascript";

        console.log(arr);

06. 다음을 보고 결괏값을 작성하시오! 

{

07. 다음을 보고 결괏값을 작성하시오! 

{

        console.log(a);
        console.log(b);
        console.log(c);
}

08. 다음을 보고 결괏값을 작성하시오! 

{

        const [a, b, [x, y, z]] = arr;

        console.log(a);
        console.log(b);
        console.log(x);
        console.log(y);
        console.log(z);
}

09. 다음을 보고 결괏값을 작성하시오! 

{
        const arr1 = [100, 200, "javascript"];
        const arr2 = [300, 400, "jquery"];
        const arr3 = [...arr1, 500];

        console.log(arr1);              
        console.log(...arr1);           
        console.log(...arr1, ...arr2);  
        console.log(...arr3);     
}

10. 다음을 보고 결괏값을 작성하시오! 

{

    console.log(...numbers); 
    
    const copiedNumbers = [...numbers];

    console.log(copiedNumbers);
}

17. 다음을 보고 결괏값을 작성하시오! 

{
        const obj = {
                a: 100,
                b: {x: 200, y: 300},
                c: [400, 500, {x: 600, y: 700}],
                d: "javascript",
                e: {x: "jquery"}
        }
        console.log(obj.b[0])
        console.log(obj.c[0])
        console.log(obj.c[2].x)
        console.log(obj.c[2].y)
        console.log(obj.e.x) 
}

18. 다음을 보고 결괏값을 작성하시오! 

{
        const obj = {
                a: 100,
                b: [200, 300],
                c: "javascript",
                d: function(){
                        console.log("javascript가 실행되었습니다.");
                },
                e: function(){
                        console.log(obj.a + "가 실행되었습니다.");
                },
                f: function(){
                        console.log(this.c + "가 실행되었습니다.");
                }
        }

        console.log(obj.a);
        console.log(obj.b[0]);
        console.log(obj.b[1]);
        console.log(obj.c);
        console.log(obj.d);
        obj.d();
        obj.e();
        obj.f();     
}

19. 다음을 보고 결괏값을 작성하시오! 

{
        const obj = {
                a: 100,
                b: {x: 200, y: 300},
                d: "javascript"
        }

        const { a, b: {x, y}, d } = obj; 

        console.log(a);
        console.log(x);
        console.log(y);
        console.log(d);
}

20. 다음을 보고 결괏값이 100이 출력되는 방법을 3가지 작성하시오.(출력은 console.log로 확인함) 

{
        const obj1 = { a: 100, b: 200, c: "javascript" };
        const obj2 = { ...obj1, d: "jquery" };
        const obj3 = { ...obj1, ...obj2 };
}
